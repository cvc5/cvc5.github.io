 18#include <iostream>

 20using namespace cvc5::api;

 22void test(Solver& slv, Sort& consListSort)
 23{
 24  // Now our old "consListSpec" is useless--the relevant information
 25  // has been copied out, so we can throw that spec away.  We can get
 26  // the complete spec for the datatype from the DatatypeSort, and
 27  // this Datatype object has constructor symbols (and others) filled in.
 28
 29  const Datatype& consList = consListSort.getDatatype();
 30
 31  // t = cons 0 nil
 32  //
 33  // Here, consList["cons"] gives you the DatatypeConstructor.  To get
 34  // the constructor symbol for application, use .getConstructor("cons"),
 35  // which is equivalent to consList["cons"].getConstructor().  Note that
 36  // "nil" is a constructor too, so it needs to be applied with
 37  // APPLY_CONSTRUCTOR, even though it has no arguments.
 38  Term t = slv.mkTerm(
 39      APPLY_CONSTRUCTOR,
 40      consList.getConstructorTerm("cons"),
 41      slv.mkInteger(0),
 42      slv.mkTerm(APPLY_CONSTRUCTOR, consList.getConstructorTerm("nil")));
 43
 44  std::cout << "t is " << t << std::endl
 45            << "sort of cons is "
 46            << consList.getConstructorTerm("cons").getSort() << std::endl
 47            << "sort of nil is " << consList.getConstructorTerm("nil").getSort()
 48            << std::endl;
 49
 50  // t2 = head(cons 0 nil), and of course this can be evaluated
 51  //
 52  // Here we first get the DatatypeConstructor for cons (with
 53  // consList["cons"]) in order to get the "head" selector symbol
 54  // to apply.
 55  Term t2 =
 56      slv.mkTerm(APPLY_SELECTOR, consList["cons"].getSelectorTerm("head"), t);
 57
 58  std::cout << "t2 is " << t2 << std::endl
 59            << "simplify(t2) is " << slv.simplify(t2) << std::endl
 60            << std::endl;
 61
 62  // You can also iterate over a Datatype to get all its constructors,
 63  // and over a DatatypeConstructor to get all its "args" (selectors)
 64  for (Datatype::const_iterator i = consList.begin(); i != consList.end(); ++i)
 65  {
 66    std::cout << "ctor: " << *i << std::endl;
 67    for (DatatypeConstructor::const_iterator j = (*i).begin(); j != (*i).end();
 68         ++j)
 69    {
 70      std::cout << " + arg: " << *j << std::endl;
 71    }
 72  }
 73  std::cout << std::endl;
 74
 75  // Alternatively, you can use for each loops.
 76  for (const auto& c : consList)
 77  {
 78    std::cout << "ctor: " << c << std::endl;
 79    for (const auto& s : c)
 80    {
 81      std::cout << " + arg: " << s << std::endl;
 82    }
 83  }
 84  std::cout << std::endl;
 85
 86  // You can also define parameterized datatypes.
 87  // This example builds a simple parameterized list of sort T, with one
 88  // constructor "cons".
 89  Sort sort = slv.mkParamSort("T");
 90  DatatypeDecl paramConsListSpec =
 91      slv.mkDatatypeDecl("paramlist",
 92                         sort);  // give the datatype a name
 93  DatatypeConstructorDecl paramCons = slv.mkDatatypeConstructorDecl("cons");
 94  DatatypeConstructorDecl paramNil = slv.mkDatatypeConstructorDecl("nil");
 95  paramCons.addSelector("head", sort);
 96  paramCons.addSelectorSelf("tail");
 97  paramConsListSpec.addConstructor(paramCons);
 98  paramConsListSpec.addConstructor(paramNil);
 99
100  Sort paramConsListSort = slv.mkDatatypeSort(paramConsListSpec);
101  Sort paramConsIntListSort =
102      paramConsListSort.instantiate(std::vector<Sort>{slv.getIntegerSort()});
103
104  const Datatype& paramConsList = paramConsListSort.getDatatype();
105
106  std::cout << "parameterized datatype sort is " << std::endl;
107  for (const DatatypeConstructor& ctor : paramConsList)
108  {
109    std::cout << "ctor: " << ctor << std::endl;
110    for (const DatatypeSelector& stor : ctor)
111    {
112      std::cout << " + arg: " << stor << std::endl;
113    }
114  }
115
116  Term a = slv.mkConst(paramConsIntListSort, "a");
117  std::cout << "term " << a << " is of sort " << a.getSort() << std::endl;
118
119  Term head_a = slv.mkTerm(
120      APPLY_SELECTOR, paramConsList["cons"].getSelectorTerm("head"), a);
121  std::cout << "head_a is " << head_a << " of sort " << head_a.getSort()
122            << std::endl
123            << "sort of cons is "
124            << paramConsList.getConstructorTerm("cons").getSort() << std::endl
125            << std::endl;
126  Term assertion = slv.mkTerm(GT, head_a, slv.mkInteger(50));
127  std::cout << "Assert " << assertion << std::endl;
128  slv.assertFormula(assertion);
129  std::cout << "Expect sat." << std::endl;
130  std::cout << "cvc5: " << slv.checkSat() << std::endl;
131}

133int main()
134{
135  Solver slv;
136  // This example builds a simple "cons list" of integers, with
137  // two constructors, "cons" and "nil."
138
139  // Building a datatype consists of two steps.
140  // First, the datatype is specified.
141  // Second, it is "resolved" to an actual sort, at which point function
142  // symbols are assigned to its constructors, selectors, and testers.
143
144  DatatypeDecl consListSpec =
145      slv.mkDatatypeDecl("list");  // give the datatype a name
146  DatatypeConstructorDecl cons = slv.mkDatatypeConstructorDecl("cons");
147  cons.addSelector("head", slv.getIntegerSort());
148  cons.addSelectorSelf("tail");
149  consListSpec.addConstructor(cons);
150  DatatypeConstructorDecl nil = slv.mkDatatypeConstructorDecl("nil");
151  consListSpec.addConstructor(nil);
152
153  std::cout << "spec is:" << std::endl << consListSpec << std::endl;
154
155  // Keep in mind that "DatatypeDecl" is the specification class for
156  // datatypes---"DatatypeDecl" is not itself a cvc5 Sort.
157  // Now that our Datatype is fully specified, we can get a Sort for it.
158  // This step resolves the "SelfSort" reference and creates
159  // symbols for all the constructors, etc.
160
161  Sort consListSort = slv.mkDatatypeSort(consListSpec);
162
163  test(slv, consListSort);
164
165  std::cout << std::endl
166            << ">>> Alternatively, use declareDatatype" << std::endl;
167  std::cout << std::endl;
168
169  DatatypeConstructorDecl cons2 = slv.mkDatatypeConstructorDecl("cons");
170  cons2.addSelector("head", slv.getIntegerSort());
171  cons2.addSelectorSelf("tail");
172  DatatypeConstructorDecl nil2 = slv.mkDatatypeConstructorDecl("nil");
173  std::vector<DatatypeConstructorDecl> ctors = {cons2, nil2};
174  Sort consListSort2 = slv.declareDatatype("list2", ctors);
175  test(slv, consListSort2);
176
177  return 0;
178}

 22  {
 23    // Now our old "consListSpec" is useless--the relevant information

 40        slv.mkTerm(Kind.APPLY_CONSTRUCTOR, consList.getConstructorTerm("nil")));
 41
 42    System.out.println("t is " + t + "\n"
 43        + "sort of cons is " + consList.getConstructorTerm("cons").getSort() + "\n"
 44        + "sort of nil is " + consList.getConstructorTerm("nil").getSort());
 45
 46    // t2 = head(cons 0 nil), and of course this can be evaluated
 47    //
 48    // Here we first get the DatatypeConstructor for cons (with
 49    // consList["cons"]) in order to get the "head" selector symbol
 50    // to apply.
 51    Term t2 =
 52        slv.mkTerm(Kind.APPLY_SELECTOR, consList.getConstructor("cons").getSelectorTerm("head"), t);
 53
 54    System.out.println("t2 is " + t2 + "\n"
 55        + "simplify(t2) is " + slv.simplify(t2) + "\n");
 56
 57    // You can also iterate over a Datatype to get all its constructors,
 58    // and over a DatatypeConstructor to get all its "args" (selectors)
 59
 60    for (Iterator<DatatypeConstructor> i = consList.iterator(); i.hasNext();)
 61    {
 62      DatatypeConstructor constructor = i.next();
 63      System.out.println("ctor: " + constructor);
 64      for (Iterator<DatatypeSelector> j = constructor.iterator(); j.hasNext();)

 67      }

 73    {

 76      {

 78      }
 79    }
 80    System.out.println();

 90    paramCons.addSelector("head", sort);
 91    paramCons.addSelectorSelf("tail");
 92    paramConsListSpec.addConstructor(paramCons);
 93    paramConsListSpec.addConstructor(paramNil);
 94
 95    Sort paramConsListSort = slv.mkDatatypeSort(paramConsListSpec);
 96    Sort paramConsIntListSort = paramConsListSort.instantiate(new Sort[] {slv.getIntegerSort()});
 97
 98    Datatype paramConsList = paramConsListSort.getDatatype();
 99
100    System.out.println("parameterized datatype sort is ");
101    for (DatatypeConstructor ctor : paramConsList)
102    {
103      System.out.println("ctor: " + ctor);
104      for (DatatypeSelector stor : ctor)
105      {
106        System.out.println(" + arg: " + stor);
107      }
108    }
109
110    Term a = slv.mkConst(paramConsIntListSort, "a");
111    System.out.println("term " + a + " is of sort " + a.getSort());
112
113    Term head_a = slv.mkTerm(
114        Kind.APPLY_SELECTOR, paramConsList.getConstructor("cons").getSelectorTerm("head"), a);
115    System.out.println("head_a is " + head_a + " of sort " + head_a.getSort() + "\n"
116        + "sort of cons is " + paramConsList.getConstructorTerm("cons").getSort() + "\n");
117    Term assertion = slv.mkTerm(Kind.GT, head_a, slv.mkInteger(50));
118    System.out.println("Assert " + assertion);
119    slv.assertFormula(assertion);
120    System.out.println("Expect sat.");
121    System.out.println("cvc5: " + slv.checkSat());
122  }
